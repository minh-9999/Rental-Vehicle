# ==================== MACRO ====================
macro(add_safe_executable target_name)
  set(sources ${ARGN})

  if(sources STREQUAL "")
    message(FATAL_ERROR "‚ùå No source files provided for target '${target_name}'")
  endif()

  foreach(file_name IN LISTS sources)
    set(full_path "${CMAKE_CURRENT_SOURCE_DIR}/${file_name}")

    if(NOT EXISTS ${full_path})
      message(FATAL_ERROR "üö´ File not found: '${file_name}' in ${CMAKE_CURRENT_SOURCE_DIR}")
    else()
      message(STATUS "‚úÖ Found source file: ${file_name}")
    endif()
  endforeach()

  add_executable(${target_name} ${sources})
  message(STATUS "üéØ Added target '${target_name}' with ${CMAKE_ARGC} source files")
endmacro()
